<?hh
/**
 * This is the main entry point file for Rider. This file brings the autoloader
 * into context, sets up the session, and allows Route to defer logic to a
 * controller based off of the request path.
 */

require_once('build/Autoloader.php');
spl_autoload_register('Autoloader::loadFile');

$user = Session::init();

Route::go($_SERVER['path'], $user);
